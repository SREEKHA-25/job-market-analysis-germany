skills = ['python', 'sql', 'excel', 'power bi', 'tableau', 'remote']

# Convert all descriptions to lowercase for easy matching
df_europe['Description'] = df_europe['Description'].fillna('').str.lower()

# Count how many job descriptions mention each skill
skill_counts = {skill: df_europe['Description'].str.contains(skill).sum() for skill in skills}

# Show as DataFrame
import pandas as pd
skill_df = pd.DataFrame.from_dict(skill_counts, orient='index', columns=['Job Count'])
skill_df.sort_values(by='Job Count', ascending=True).plot(kind='barh', figsize=(6, 4), color='orchid')

plt.title("Demand for Skills in European Job Listings")
plt.xlabel("Number of Job Listings")
plt.tight_layout()
plt.show()
